Logic

  • opening()
    • SET greeting to "Welcome to Bank of Satch!"
    • DISPLAY greeting
    • DISPLAY "Please Enter Card"
    • checkCard()
  • checkCard()
    • SET card to zero
    • INPUT card
    • IF card is true:
      • checkPIN()
    • ELSE IF card is false:
      • DISPLAY "Invalid card"
      • END
  • checkPIN()
    • DISPLAY "Enter Your 4-Digit pin"
    • SET pin to zero
    • SET pinCheck to 1234
    • INPUT pin
    • IF pin is equal to pinCheck:
      • optionScreen()
    • ELSE:
      • pinAttempt += 1
      • DISPLAY "Invalid PIN. Please Try Again."
      • LOOP checkPIN()
    • IF pinAttempt = 3:
      • DISPLAY "Maximum Amount of Tries
      • END
  • optionScreen()
    • DISPLAY "Please Select an Option"
    • DISPLAY "Make Deposit", "Make Withdrawal", "View Balance"
  • IF "Make Deposit":
    • DISPLAY "Click Savings or Checking Deposit"
    • IF "Savings"
      • savingsDeposit()
      • DISPLAY "How much is the deposit?"
      • INPUT depositAmount
      • SET savingsBalance = depositAmount + savingsBalance
      • DISPLAY "Here is your new balance:" + savingsBalance
    • ELSE "Checking"
      • checkingDeposit()
      • DISPLAY "How much is the deposit?"
      • INPUT depositAmount
      • SET checkingBalance = depositAmount + checkingBalance
      • DISPLAY "Here is your new balance:" + checkingBalance
  • ELSE IF "Make Withdrawal"
    • DISPLAY "From Savings or Checking?"
    • IF "Savings"
      • savingsWithdrawal()
      • DISPLAY "How much would you like to withdraw?
      • INPUT savingsWithdrawal
      • IF savingsWithdrawal > withdrawLimit:
        • DISPLAY "Excessive Amount"
        • LOOP savingsWithdrawal
      • ELSE IF savingsWithdrawal < withdrawLimit
      • IF savingsWithdrawal < savingsBalance:
        • DISPLAY "Dispensing" + savingsWithdrawal
        • DISPLAY "Would you like another transaction?"
        • IF "Yes"
          • LOOP to optionScreen()
        • ELSE "No"
          • END
      • ELSE IF savingsWithdrawal > savingsBalance:
        • DISPLAY "You have insufficient funds for this transaction."
        • DISPLAY "Choose Another Transaction or END"
          • IF "Another Transaction":
          • LOOP to optionScreen()
          • ELSE IF "END"
            • END
    • ELSE IF "Checking"
      • checkingWithdrawal()
      • DISPLAY "How much would you like to withdraw?
      • INPUT checkingWithdrawal
      • IF checkingWithdrawal < checkingBalance:
        • DISPLAY "Dispensing" + checkingWithdrawal
        • DISPLAY "Would you like another transaction?"
        • IF "Yes"
          • LOOP to optionScreen()
        • ELSE "No"
          • END
      • ELSE IF checkingWithdrawal > checkingBalance:
        • DISPLAY "You have insufficient funds for this transaction."
        • DISPLAY "Choose Another Transaction or END"
          • IF "Another Transaction":
          • LOOP to optionScreen()
          • ELSE IF "END"
            • END
  • IF "View Balance":
    • checkBalance()
    • DISPLAY "Savings or Checking?"
    • IF "Savings"
      • DISPLAY savingsBalance
      • LOOP to optionScreen()
    • ELSE IF "Checking"
      • DISPLAY checkingBalance
      • LOOP to optionScreen()
